    """
    A Whetlab tuning experiment.

    :ivar parameters: Parameters to be tuned during the experiment.
    :type parameters: dict
    :ivar outcome: Description of the outcome to minimize.
    :type outcome: dict
    """
    
    def __init__(self,
                 parameters,
                 outcome):
        """
        Experiment instance constructor.

        The parameters to tune in the experiment are specified by
        ``parameters``. It should be a ``dict``, where keys are
        the parameters (``str``) and values are ``dict`` that
        provide information about these parameters. Each of these
        ``dict`` should the appropriate keys to properly describe
        the parameter:
        
        * ``'type'``: type of the parameter (default: ``'float'``)
        * ``'min'``: minimum value of the parameter
        * ``'max'``: maximum value of the parameter
        * ``'size'``: size of parameter (default: ``1``)
        * ``'units'``: units (``str``) in which the parameter is measured
        * ``'scale'``: scale to use when exploring parameter values (default: ``'linear'``)

        Outcome should also be a ``dict``, describing the outcome. It
        should have the keys:

        * ``'name'``: name (``str``) for the outcome being optimized
        * ``'type'``: type of the outcome (default: ``'float'``)

        :param parameters: Parameters to be tuned during the experiment.
        :type parameters: dict
        :param outcome: Description of the outcome to minimize.
        :type outcome: dict

        parameters is a dict of dict
        (must mention the supported types : float, int, enum)
        """
    
        self.parameters = collections.OrderedDict(parameters)
        self.outcome = outcome

        # Temporary: initialize the results for reporting and to use the backend
        self._next_id = 0
        self._param_values = collections.OrderedDict({})
        self._outcome_values = collections.OrderedDict({})
        self._pv_to_ids = collections.OrderedDict({})

        # Setup REST server communcation
        # Temporary
        # Construct the config file
        self._options = dict()
        self._options['config'] = parameters

        # Default configuration options

        # Temporary
        best,argbest = np.inf,None
        for i in self._outcome_values.iteritems():
            current = i[1]['val']
            if current < best:
                argbest = i[0]
                best = current
        return argbest
    


[docs]    def report(self):
        """
        Plot a visual report of the progress made so far in the experiment.
        """

        # Report historical progress and experiments assumed pending
        import matplotlib.pyplot as plt

        # Plot progression
        plt.figure(1)
        plt.clf()
        y = [ x['val'] for x in self._outcome_values.itervalues() if x['val'] is not 'P']
        best_so_far = [ np.min(y[:(i+1)]) for i in range(len(y)) ]
        plt.scatter(range(len(y)),y,marker='x',color='k',label='Outcomes')
        plt.plot(range(len(y)),best_so_far,color='k',label='Best so far')
        plt.xlabel('Experiment ID')
        plt.ylabel(self.outcome['name'])
        plt.title('Outcome values progression')
        plt.legend()
        plt.draw()
        plt.ion()
        plt.show()
        
        # Plot table of experiments
        plt.figure(2)
        param_names = self.parameters.keys()
        col_names = param_names + [self.outcome['name']]
        cell_text = []
        for k,v in self._param_values.iteritems():
            outcome = self.outcome_values[k]
            cell_text.append([str(v[n]) for n in param_names] + [str(outcome['val'])])
        plt
        the_table = plt.table(cellText = cell_text, colLabels=col_names, loc='center')
        plt.axis('off')
        plt.title('Table of experiments')
        plt.draw()
        plt.ion()
        plt.show()
